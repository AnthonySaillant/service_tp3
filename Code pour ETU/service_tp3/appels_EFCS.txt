## Consulter un film en particulier
query {
  film(id: 1) {
    title
    critics {
      id
      score
      comment
    }
  }
}

## Consultation de tous les acteurs d’un certain film
query {
  film(id: 1) {
    title
    actors {
      last_name
      first_name
      birthdate
    }
  }
}

## Recherche de films - Cas 1 : Sans critère avec pagination (demandez la page 1)
query {
  films(page: 1, first: 10) {
    paginatorInfo {
      currentPage
    }
    data {
      id
      title
      release_year
      length
      description
    }
  }
}

## Recherche de films - Cas 2 : Avec le mot-clé "AL" (sur titre ou descritpion) avec pagination (demandez la page 1)
query {
  films(
    page: 10
    first: 10
    keywordTitle: "AL"
    keywordDescription: "AL"
  ) {
    paginatorInfo {
      currentPage
    }
    data {
      id
      title
      release_year
      length
      description
    }
  }
}

## Recherche de films - Cas 3 : Avec le mot-clé "AL" (sur titre ou descritpion), minimum année 2006, entre 120 et 150 minutes avec pagination (demandez la page 1)
query {
  films(
    page: 1
    first: 10
    keywordTitle: "AL"
    keywordDescription: "AL"
    minYear: 2006
    minLength: 120
    maxLength: 150
  ) {
    paginatorInfo {
      currentPage
    }
    data {
      id
      title
      release_year
      length
      description
    }
  }
}

## Ajout d’une critique de film
mutation {
  createCritic(input: {
    film_id: 2,
    score: 100,
    comment: "Ouah le film de dinguos!!!"
  }) {
    id
    score
    comment
    film {
      title
      statistic {
        average_score
        nb_votes
      }
    }
  }
}
